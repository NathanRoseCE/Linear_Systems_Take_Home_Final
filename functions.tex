
% includes a formatted image with a figure tag and caption
% 1: image filepath
% 2: Caption
% 3: label tag
\newcommand\image[3]{
  \begin{figure}[H]
    \begin{center}
      \makebox[\textwidth]{\includegraphics[max width=\textwidth]{#1}}
    \end{center}
    \caption{#2}
    \label{#3}
  \end{figure}
}
